MethodDescription

Data Format

{
  nodes: [{
    id: number | string,
    labels: string[],
    data: Record<string, any>,
    visible: boolean,
    color: string,
    size?: number,           // Radius in world units (default: 9)
    expand?: boolean,
  }],
  links: [{
    id: number | string,
    source: number | string,
    target: number | string,
    relationship: string,
    data: Record<string, any>,
    color: string,
    visible: boolean,
  }]
}
